    def __init__(self, initial_id, pos):
        self.__i = np.linspace(0, 1)
        self.__one = np.ones_like(self.__i)
        self.__x = [np.copy(self.__i)]
        self.__y = [pos * self.__one]

        self.__color = "black"
        self.__label = ""

    def __repr__(self):

        return self.__y[-1][-1]

    def add_identity(self):
        self.__x.append(self.__i + self.get_last_x())
